// Anti-close system - Add this script to any page where you want the protection (function() { // Main functions for handling anti-close behavior function setupAntiClose() { window.addEventListener('beforeunload', showWarning); updateActiveState(true); } function removeAntiClose() { window.removeEventListener('beforeunload', showWarning); updateActiveState(false); } function showWarning(e) { // Don't show warning for redirects or if system is disabled if (!e.target.location.href.includes("redirect") && localStorage.getItem('anticlose') === 'true') { e.preventDefault(); e.returnValue = 'Leave site? Changes you made may not be saved.'; return e.returnValue; } } // Update active state and handle UI if present function updateActiveState(isActive) { // Update checkbox if it exists on the current page const checkbox = document.getElementById('cbox_anticlose'); if (checkbox) { checkbox.checked = isActive; } } // Initialize anti-close system function initAntiClose() { // Check if anti-close is enabled in localStorage const isEnabled = localStorage.getItem('anticlose') === 'true'; // Set up initial state if (isEnabled) { setupAntiClose(); } else if (localStorage.getItem('anticlose') === null) { // First time initialization localStorage.setItem('anticlose', 'false'); } // Update UI to match current state updateActiveState(isEnabled); } // Listen for changes from other tabs/windows window.addEventListener('storage', function(e) { if (e.key === 'anticlose') { if (e.newValue === 'true') { setupAntiClose(); } else { removeAntiClose(); } } }); // Initialize when DOM is ready if (document.readyState === 'loading') { document.addEventListener('DOMContentLoaded', initAntiClose); } else { initAntiClose(); } })(); New Tab
